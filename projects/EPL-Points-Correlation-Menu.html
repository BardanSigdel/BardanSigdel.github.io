#Defining functions to display the menu

def display_menu():
    print("Menu for Selection:")
    print()
    print("1. Display Summary Stats for Premier League clubs Expenditure and Points")
    print("2. Display Mean Points by Location")
    print("3. Display Scatter Plot of Expenditure and Points")
    print("4. Display Bar Chart of Mean Points by Club Location")
    print("5. Exit Menu")
    print()
    choice = input("Enter your menu choice: ")
    return int(choice)

#Defining scatter plot

def scatter_plot():
    plt.scatter(final_df['Expenditure'], final_df['Points'])
    plt.xlabel('Expenditure, in millions')
    plt.ylabel('Total Points')
    plt.title('Scatter Plot of Expenditure vs Total Points')
    plt.show()

#Defining Bar Chart

def bar_chart():
    plt.bar(final_df['Location'], final_df['Points'])
    plt.xticks(rotation=90)
    plt.xlabel('Location')
    plt.ylabel('Total Points')
    plt.title('Bar Chart of Total Points by Location')
    plt.show()

#Running Menu

def main():
    choice = 0
    while choice != 5:
        choice = display_menu()
        if choice == 1:
            summary_stats()
        elif choice == 2:
            mean_points_by_loc()
        elif choice == 3:
            scatter_plot()
        elif choice == 4:
            bar_chart()
        elif choice == 5: 
            print("You exited from the menu")
        else:
            print("Invalid menu choice, try again")
